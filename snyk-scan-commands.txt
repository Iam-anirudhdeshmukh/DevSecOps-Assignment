# Build the Docker image
docker build -t devsecops-python-app:latest .

# Create results folder
mkdir -p results

# Run Snyk container scan
snyk container test devsecops-python-app:latest \
  --experimental \
  --json-file-output=results/snyk-report.json

# (Optional) Show summary using jq
jq '.vulnerabilities[] | {severity, title, packageName}' results/snyk-report.json
